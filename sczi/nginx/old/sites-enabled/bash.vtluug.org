server {
    listen 80;
    listen [::]:80;
    server_name bash.vtluug.org;

    location /.well-known/acme-challenge {
        root /var/lib/letsencrypt/public_html;
        default_type text/plain;
    }
    
    location / {
    	return 301 https://bash.vtluug.org$request_uri;
    }
}
